        // Atualizar classe de prioridade

        function updatePriorityClass(cell) {

            cell.classList.remove('priority-high', 'priority-medium', 'priority-low');

            const priority = cell.textContent.toLowerCase();

            

            if (priority.includes('alta')) {

                cell.classList.add('priority-high');

            } else if (priority.includes('média') || priority.includes('media')) {

                cell.classList.add('priority-medium');

            } else if (priority.includes('baixa')) {

                cell.classList.add('priority-low');

            }

        }
